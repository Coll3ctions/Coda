    else()
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${wlevel}")
    endif()

    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /Oi /Ob2")

            "fi\n"

            "${LCOV_PATH} --directory . --capture --output-file coverage.info\n"

    endif()
